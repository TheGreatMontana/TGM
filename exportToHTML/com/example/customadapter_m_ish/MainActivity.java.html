12   import android.widget.Toast;
13   
14   import java.util.ArrayList;
15   import java.util.List;
16   
17   public class MainActivity extends Activity {
18       private ListView lvProduct;
19       private ProductListAdapter adapter;
20       private List<Product> mProductList;
21       private Button btnAddProduct;
22   
23       @Override
24       protected void onCreate(Bundle savedInstanceState) {
25           super.onCreate(savedInstanceState);
26           setContentView(R.layout.activity_main);
27   
28           lvProduct = findViewById(R.id.listview_product);
29           btnAddProduct = findViewById(R.id.btn_add_product);
30   
31           mProductList = new ArrayList<>();
32           mProductList.add(new Product(1, "Iphone 15", 899, "Apple Iphone 15 128GB"));
33           mProductList.add(new Product(2, "Iphone 15 Plus", 999, "Apple Iphone 15 Plus 128GB"));
34           mProductList.add(new Product(3, "Iphone 15 Pro", 1099, "Apple Iphone 15 Pro 128GB"));
35           mProductList.add(new Product(4, "Iphone 15 Pro Max", 1199, "Apple Iphone 15 Pro Max 256GB"));
36   
37           adapter = new ProductListAdapter(getApplicationContext(), mProductList);
38           lvProduct.setAdapter(adapter);
39   
40           lvProduct.setOnItemClickListener(new AdapterView.OnItemClickListener() {
41               @Override
42               public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
43                   Toast.makeText(getApplicationContext(), "Bosildi: " + mProductList.get(position).getName(), Toast.LENGTH_SHORT).show();
44               }
45           });
46   
47           btnAddProduct.setOnClickListener(new View.OnClickListener() {
48               @Override
49               public void onClick(View v) {
50                   showAddProductDialog();
51               }
52           });
53       }
54   
55       private void showAddProductDialog() {
56           AlertDialog.Builder builder = new AlertDialog.Builder(this);
57           View dialogView = getLayoutInflater().inflate(R.layout.dialog_add_product, null);
58   
59           EditText etName = dialogView.findViewById(R.id.et_product_name);
60           EditText etPrice = dialogView.findViewById(R.id.et_product_price);
61           EditText etDescription = dialogView.findViewById(R.id.et_product_description);
62   
63           builder.setView(dialogView)
64                   .setTitle("Yangi tovar qo'shish")
65                   .setPositiveButton("Qo'shish", new DialogInterface.OnClickListener() {
66                       @Override
67                       public void onClick(DialogInterface dialog, int which) {
68                           String name = etName.getText().toString();
69                           String priceStr = etPrice.getText().toString();
70                           String description = etDescription.getText().toString();
71   
72                           if (!name.isEmpty() && !priceStr.isEmpty() && !description.isEmpty()) {
73                               int price = Integer.parseInt(priceStr);
74                               mProductList.add(new Product(mProductList.size() + 1, name, price, description));
75                               adapter.notifyDataSetChanged();
76                           } else {
77                               Toast.makeText(MainActivity.this, "Iltimos barcha qatorlarni to'ldiring!", Toast.LENGTH_SHORT).show();
78                           }
79                       }
80                   })
81                   .setNegativeButton("Bekor qilish", null)
82                   .create()
83                   .show();
84       }
85   }
